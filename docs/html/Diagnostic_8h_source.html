186  bool enable_diagnostics(std::vector<Diagnostic::DiagnosticType> diagnostic_types) {
187  if (initialized == false) {
188  return false;
189  }
190  std::sort(diagnostic_types.begin(), diagnostic_types.end());
191  for (std::size_t i = 0; i < diagnostic_types.size(); ++i) {
192  DiagnosticDefinition diag = root_diagnostic;
193  diag.type = diagnostic_types.at(i);
194  if (diag.type ==
195  Diagnostic::DiagnosticType::SYSTEM_RESOURCE) // This is special, so we don't throw
196  // a ton of warn messages when the
197  // system launches.
198  {
199  diag.level = Level::Type::NOTICE;
200  diag.message = Diagnostic::Message::INITIALIZING;
201  diag.description = "Initializing Resource Monitor.";
202  }
203  else {

205  diag.message = Diagnostic::Message::INITIALIZING;
206  diag.description = "Initializing Diagnostic.";
207  }

209  }

245  DiagnosticDefinition update_diagnostic(std::string device_name,
246  Diagnostic::DiagnosticType diagnostic_type,
247  Level::Type level,
248  Diagnostic::Message message,
249  std::string description) {
250  bool devicetype_found = false;
251  bool devicename_found = false;
252  DiagnosticDefinition diag;
253  uint8_t insert_index = -1;
254  for (std::size_t i = 0; i < diagnostics.size(); ++i) {
255  if (diagnostic_type == diagnostics.at(i).type) {
256  devicetype_found = true;
257  insert_index = i;
258  if (diagnostics.at(i).device_name == device_name) {
259  devicename_found = true;
260  diag = diagnostics.at(i);
261  diag.level = level;
262  diag.message = message;
263  diag.description = description;
264  diagnostics.at(i) = diag;
265  }
266  }
267  }
268  if ((devicetype_found == true) and (devicename_found == false)) {
269  diag = root_diagnostic;
270  diag.type = diagnostic_type;
271  diag.device_name = device_name;
272  diag.level = level;
273  diag.message = message;
274  diag.description = description;
275  std::vector<DiagnosticDefinition>::iterator it;
276  it = diagnostics.begin();
277  diagnostics.insert(it + insert_index, diag);
278  }
279  if (devicetype_found == true) {
280  return diag;
281  }
282  else {
283  diag = root_diagnostic;
284  diag.type = diagnostic_type;
285  diag.level = Level::Type::ERROR;
286  diag.message = Diagnostic::Message::INITIALIZING_ERROR;
287  char tempstr[512];
288  sprintf(tempstr,
289  "Unsupported Diagnostic Type: %s(%d). Did you forget to enable it?",
290  DiagnosticTypeString(diagnostic_type).c_str(),
291  (uint8_t)diagnostic_type);
292  diag.description = std::string(tempstr);
293  return diag;
294  }
295  }
